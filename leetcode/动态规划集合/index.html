int edit_distance(vector<char> str1, vector<char> str2){
    int len1 = str1.size();
    int len2 = str2.size();
    if(len1==0 || len2==0)
        return max(len1, len2);
    vector<vector<int>> dp(len1+1, vector<int>(len2+1, 0));
    for(int i=0; i<=len1; i++)
        dp[i][0]=i;
    for(int j=0; j<=len2;j++)
        dp[0][j]=j;
    for(int i=1; i<=len1; i++)
        for(int j=1;j<=len2; j++){

            }
        }
    return dp[len1][len2];
}
