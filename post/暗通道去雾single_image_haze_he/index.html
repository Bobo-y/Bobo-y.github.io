import cv2
import math
import numpy as np


def dark_channel(im, sz):
    # æš—é€šé“è®¡ç®—
    b, g, r = cv2.split(im)
    # å–æ¯ä¸ªåƒç´ ä½ç½®RGBä¸‰é€šé“ä¸­çš„æœ€å°å€¼

    kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (sz, sz))
    # è¿›è¡Œæœ€å°å€¼æ»¤æ³¢
    dark = cv2.erode(dc, kernel)
    return dark


def atm_light(im, dark):
    # è®¡ç®—å…¨çƒå¤§æ°”å…‰çš„å€¼
    [h, w] = im.shape[:2]
    imsz = h * w
    numpx = int(max(math.floor(imsz / 1000), 1))
    darkvec = dark.reshape(imsz, 1)
    imvec = im.reshape(imsz, 3)

    indices = darkvec.argsort()
    indices = indices[imsz - numpx::]

    atmsum = np.zeros([1, 3])
    # ä½¿ç”¨å‰0.1%çš„å‡å€¼, è€Œä¸æ˜¯æœ€å¤§å€¼
    for ind in range(1, numpx):
        atmsum = atmsum + imvec[indices[ind]]

    A = atmsum / numpx
    return A


def transmission_estimate(im, A, sz):
    # æŒ‰ç…§å…¬å¼ï¼ˆ12ï¼‰è®¡ç®—
    omega = 0.95
    im3 = np.empty(im.shape, im.dtype)
    for ind in range(0, 3):
        im3[:, :, ind] = im[:, :, ind] / A[0, ind]

    transmission = 1 - omega * dark_channel(im3, sz)
    return transmission


def guided_filter(im, p, r, eps):
    # http://kaiminghe.com/eccv10/index.html
    mean_I = cv2.boxFilter(im, cv2.CV_64F, (r, r))
    mean_p = cv2.boxFilter(p, cv2.CV_64F, (r, r))
    mean_Ip = cv2.boxFilter(im * p, cv2.CV_64F, (r, r))
    cov_Ip = mean_Ip - mean_I * mean_p

    mean_II = cv2.boxFilter(im * im, cv2.CV_64F, (r, r))
    var_I = mean_II - mean_I * mean_I

    a = cov_Ip / (var_I + eps)
    b = mean_p - a * mean_I

    mean_a = cv2.boxFilter(a, cv2.CV_64F, (r, r))
    mean_b = cv2.boxFilter(b, cv2.CV_64F, (r, r))

    q = mean_a * im + mean_b
    return q


def transmission_refine(im, et):
    gray = cv2.cvtColor(im, cv2.COLOR_BGR2GRAY)
    gray = np.float64(gray) / 255
    r = 60
    eps = 0.0001
    t = guided_filter(gray, et, r, eps)
    return t


def recover(im, t, A, tx=0.1):
    # å…¬å¼(16)
    res = np.empty(im.shape, im.dtype)
    t = cv2.max(t, tx)
    for ind in range(0, 3):
        res[:, :, ind] = (im[:, :, ind] - A[0, ind]) / t + A[0, ind]
    return res


if __name__ == '__main__':

    src = cv2.imread("1.png")
    I = src.astype('float64') / 255
    dark = dark_channel(I, 15)
    A = atm_light(I, dark)
    te = transmission_estimate(I, A, 15)
    t = transmission_refine(src, te)
    J = recover(I, t, A, 0.1)

    cv2.imshow("dark", dark)
    cv2.imshow("t", t)
    cv2.imshow('I', src)
    cv2.imshow('J', J)
    cv2.imwrite("J.png", J * 255)
